24  package oshi.util;
25  
26  import java.io.BufferedReader;
27  import java.io.IOException;
28  import java.io.InputStreamReader;
29  import java.nio.charset.Charset;
30  import java.util.ArrayList;
31  import java.util.Arrays;
32  import java.util.List;
33  
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  import oshi.annotation.concurrent.ThreadSafe;
38  
39  /**
40   * A class for executing on the command line and returning the result of
41   * execution.
42   */
43  @ThreadSafe
44  public final class ExecutingCommand {
45  
46      private static final Logger LOG = LoggerFactory.getLogger(ExecutingCommand.class);
47  
48      private ExecutingCommand() {
49      }
50  
51      /**
52       * Executes a command on the native command line and returns the result. This is
53       * a convenience method to call {@link java.lang.Runtime#exec(String)} and
54       * capture the resulting output in a list of Strings. On Windows, built-in
55       * commands not associated with an executable program may require
56       * {@code cmd.exe /c} to be prepended to the command.
57       *
58       * @param cmdToRun
59       *            Command to run
60       * @return A list of Strings representing the result of the command, or empty
61       *         string if the command failed
62       */
63      public static List<String> runNative(String cmdToRun) {
64          String[] cmd = cmdToRun.split(" ");
65          return runNative(cmd);
66      }
67  
68      /**
69       * Executes a command on the native command line and returns the result line by
70       * line. This is a convenience method to call
71       * {@link java.lang.Runtime#exec(String[])} and capture the resulting output in
72       * a list of Strings. On Windows, built-in commands not associated with an
73       * executable program may require the strings {@code cmd.exe} and {@code /c} to
74       * be prepended to the array.
75       *
76       * @param cmdToRunWithArgs
77       *            Command to run and args, in an array
78       * @return A list of Strings representing the result of the command, or empty
79       *         string if the command failed
80       */
81      public static List<String> runNative(String[] cmdToRunWithArgs) {
82          Process p = null;
83          try {
84              p = Runtime.getRuntime().exec(cmdToRunWithArgs);
85          } catch (SecurityException | IOException e) {
86              LOG.trace("Couldn't run command {}: {}", Arrays.toString(cmdToRunWithArgs), e.getMessage());
87              return new ArrayList<>(0);
88          }
89  
90          ArrayList<String> sa = new ArrayList<>();
91          try (BufferedReader reader = new BufferedReader(
92                  new InputStreamReader(p.getInputStream(), Charset.defaultCharset()))) {
93              String line;
94              while ((line = reader.readLine()) != null) {
95                  sa.add(line);
96              }
97              p.waitFor();
98          } catch (IOException e) {
99              LOG.trace("Problem reading output from {}: {}", Arrays.toString(cmdToRunWithArgs), e.getMessage());
100             return new ArrayList<>(0);
101         } catch (InterruptedException ie) {
102             LOG.trace("Interrupted while reading output from {}: {}", Arrays.toString(cmdToRunWithArgs),
103                     ie.getMessage());
104             Thread.currentThread().interrupt();
105         }
106         return sa;
107     }
108 
109     /**
110      * Return first line of response for selected command.
111      *
112      * @param cmd2launch
113      *            String command to be launched
114      * @return String or empty string if command failed
115      */
116     public static String getFirstAnswer(String cmd2launch) {
117         return getAnswerAt(cmd2launch, 0);
118     }
119 
120     /**
121      * Return response on selected line index (0-based) after running selected
122      * command.
123      *
124      * @param cmd2launch
125      *            String command to be launched
126      * @param answerIdx
127      *            int index of line in response of the command
128      * @return String whole line in response or empty string if invalid index or
129      *         running of command fails
130      */
131     public static String getAnswerAt(String cmd2launch, int answerIdx) {
132         List<String> sa = ExecutingCommand.runNative(cmd2launch);
133 
134         if (answerIdx >= 0 && answerIdx < sa.size()) {
135             return sa.get(answerIdx);
136         }
137         return "";
138     }
139 
140 }
