150     /**
151      * Contains information about a Remote Desktop Services session.
152      */
153     @FieldOrder({ "State", "SessionId", "IncomingBytes", "OutgoingBytes", "IncomingFrames", "OutgoingFrames",
154             "IncomingCompressedBytes", "OutgoingCompressedBytes", "WinStationName", "Domain", "UserName", "ConnectTime",
155             "DisconnectTime", "LastInputTime", "LogonTime", "CurrentTime" })
156     class WTSINFO extends Structure {
157         private static final int CHAR_WIDTH = Boolean.getBoolean("w32.ascii") ? 1 : 2;
158 
159         public int State; // WTS_CONNECTSTATE_CLASS
160         public int SessionId;
161         public int IncomingBytes;
162         public int OutgoingBytes;
163         public int IncomingFrames;
164         public int OutgoingFrames;
165         public int IncomingCompressedBytes;
166         public int OutgoingCompressedBytes;
167         public final byte[] WinStationName = new byte[WINSTATIONNAME_LENGTH * CHAR_WIDTH];
168         public final byte[] Domain = new byte[DOMAIN_LENGTH * CHAR_WIDTH];
169         public final byte[] UserName = new byte[(USERNAME_LENGTH + 1) * CHAR_WIDTH];
170         public LARGE_INTEGER ConnectTime;
171         public LARGE_INTEGER DisconnectTime;
172         public LARGE_INTEGER LastInputTime;
173         public LARGE_INTEGER LogonTime;
174         public LARGE_INTEGER CurrentTime;
175 
176         public WTSINFO() {
177             super();
178         }
179 
180         public WTSINFO(Pointer p) {
181             super(p);
182             read();
183         }
184 
185         /**
186          * Convenience method to return the null-terminated string in the
187          * {@link #WinStationName} member, accounting for {@code CHAR} or {@code WCHAR}
188          * byte width.
189          *
190          * @return The {@code WinStationName} as a string.
191          */
192         public String getWinStationName() {
193             return getStringAtOffset(fieldOffset("WinStationName"));
194         }
195 
196         /**
197          * Convenience method to return the null-terminated string in the
198          * {@link #Domain} member, accounting for {@code CHAR} or {@code WCHAR} byte
199          * width.
200          *
201          * @return The {@code Domain} as a string.
202          */
203         public String getDomain() {
204             return getStringAtOffset(fieldOffset("Domain"));
205         }
206 
207         /**
208          * Convenience method to return the null-terminated string in the
209          * {@link #UserName} member, accounting for {@code CHAR} or {@code WCHAR} byte
210          * width.
211          *
212          * @return The {@code UserName} as a string.
213          */
214         public String getUserName() {
215             return getStringAtOffset(fieldOffset("UserName"));
216         }
217 
218         private String getStringAtOffset(int offset) {
219             return CHAR_WIDTH == 1 ? getPointer().getString(offset) : getPointer().getWideString(offset);
220         }
221     }

223     /**
224      * Retrieves a list of sessions on a Remote Desktop Session Host (RD Session
225      * Host) server.
226      *
227      * @param hServer
228      *            A handle to the RD Session Host server.
229      *            <p>
230      *            You can use the {@code WTSOpenServer} or {@code WTSOpenServerEx}
231      *            functions to retrieve a handle to a specific server, or
232      *            {@link #WTS_CURRENT_SERVER_HANDLE} to use the RD Session Host
233      *            server that hosts your application.
234      * @param Reserved
235      *            This parameter is reserved. It must be zero.
236      * @param Version
237      *            The version of the enumeration request. This parameter must be 1.
238      * @param ppSessionInfo
239      *            A pointer to an array of {@link WTS_SESSION_INFO} structures that
240      *            represent the retrieved sessions. To free the returned buffer,
241      *            call the {@link Wtsapi32#WTSFreeMemory} function.
242      * @param pCount
243      *            A pointer to the number of {@code WTS_SESSION_INFO} structures
244      *            returned in the {@code ppSessionInfo} parameter.
245      * @return Returns {@code false} if this function fails. If this function
246      *         succeeds, returns {@code true}.
247      *         <p>
248      *         To get extended error information, call
249      *         {@link Kernel32#GetLastError()}.
250      */
251     boolean WTSEnumerateSessions(HANDLE hServer, int Reserved, int Version, PointerByReference ppSessionInfo,
252             IntByReference pCount);
253 
254     /**
255      * Retrieves session information for the specified session on the specified
256      * Remote Desktop Session Host (RD Session Host) server. It can be used to query
257      * session information on local and remote RD Session Host servers.
258      *
259      * @param hServer
260      *            A handle to an RD Session Host server. Specify a handle opened by
261      *            the {@code WTSOpenServer} function, or specify
262      *            {@link #WTS_CURRENT_SERVER_HANDLE} to indicate the RD Session Host
263      *            server on which your application is running.
264      * @param SessionId
265      *            A Remote Desktop Services session identifier. To indicate the
266      *            session in which the calling application is running (or the
267      *            current session) specify {@link #WTS_CURRENT_SESSION}. Only
268      *            specify {@code WTS_CURRENT_SESSION} when obtaining session
269      *            information on the local server. If {@code WTS_CURRENT_SESSION} is
270      *            specified when querying session information on a remote server,
271      *            the returned session information will be inconsistent. Do not use
272      *            the returned data.
273      *            <p>
274      *            You can use the {@code WTSEnumerateSessionsEx} function to
275      *            retrieve the identifiers of all sessions on a specified RD Session
276      *            Host server.
277      *            <p>
278      *            To query information for another user's session, you must have
279      *            Query Information permission.
280      * @param WTSInfoClass
281      *            A value of the {@link WTS_INFO_CLASS} enumeration that indicates
282      *            the type of session information to retrieve in a call to the
283      *            {@code WTSQuerySessionInformation} function.
284      * @param ppBuffer
285      *            A pointer to a variable that receives a pointer to the requested
286      *            information. The format and contents of the data depend on the
287      *            information class specified in the {@code WTSInfoClass} parameter.
288      *            To free the returned buffer, call the {@link #WTSFreeMemory}
289      *            function.
290      * @param pBytesReturned
291      *            A pointer to a variable that receives the size, in bytes, of the
292      *            data returned in ppBuffer.
293      * @return If the function succeeds, returns {@code true}.
294      *         <p>
295      *         If the function fails, returns {@code false}. To get extended error
296      *         information, call {@link Kernel32#GetLastError()}.
297      */
298     boolean WTSQuerySessionInformation(HANDLE hServer, int SessionId, int WTSInfoClass, PointerByReference ppBuffer,
299             IntByReference pBytesReturned);
