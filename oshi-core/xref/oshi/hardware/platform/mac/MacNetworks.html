54      private static final int CTL_NET = 4;

57      private static final int RTM_IFINFO2 = 0x12;

59      @Override
60      public NetworkIF[] getNetworks() {
61          List<NetworkIF> result = new ArrayList<>();
62          Map<Integer, IFdata> ifDataMap = queryIFdata(-1);
63  
64          for (NetworkInterface netint : getNetworkInterfaces()) {
65              NetworkIFetworkIF">NetworkIF netIF = new NetworkIF();
66              netIF.setNetworkInterface(netint);
67              updateNetworkStats(netIF, ifDataMap);
68              result.add(netIF);
69          }
70  
71          return result.toArray(new NetworkIF[0]);
72      }

74      /**
75       * Map all network interfaces. Ported from source code of "netstat -ir". See
76       * http://opensource.apple.com/source/network_cmds/network_cmds-457/
77       * netstat.tproj/if.c
78       *
79       * @param index
80       *            If nonnegative, returns results for only that index
81       * @return a map of {@link IFData} object indexed by the interface index,
82       *         encapsulating the stats, or {@code null} if the query failed.
83       */
84      private static Map<Integer, IFdata> queryIFdata(int index) {
85          Map<Integer, IFdata> data = new HashMap<>();
86          // Get buffer of all interface information
87          int[] mib = { CTL_NET, PF_ROUTE, 0, 0, NET_RT_IFLIST2, 0 };
88          IntByReference len = new IntByReference();
89          if (0 != SystemB.INSTANCE.sysctl(mib, 6, null, len, null, 0)) {
90              LOG.error("Didn't get buffer length for IFLIST2");
91              return data;
92          }
93          Memory buf = new Memory(len.getValue());
94          if (0 != SystemB.INSTANCE.sysctl(mib, 6, buf, len, null, 0)) {
95              LOG.error("Didn't get buffer for IFLIST2");
96              return data;
97          }
98  
99          // Iterate offset from buf's pointer up to limit of buf
100         int lim = (int) (buf.size() - new IFmsgHdr().size());
101         int offset = 0;
102         while (offset < lim) {
103             // Get pointer to current native part of buf
104             Pointer p = buf.share(offset);
105             // Cast pointer to if_msghdr
106             IFmsgHdr ifm = new IFmsgHdr(p);
107             ifm.read();
108             // Advance next
109             offset += ifm.ifm_msglen;
110             // Skip messages which are not the right format
111             if (ifm.ifm_type == RTM_IFINFO2) {
112                 // Cast pointer to if_msghdr2
113                 IFmsgHdr2 if2m = new IFmsgHdr2(p);
114                 if2m.read();
115                 if (index < 0 || index == if2m.ifm_index) {
116                     data.put((int) if2m.ifm_index,
117                             new IFdata(if2m.ifm_data.ifi_type, if2m.ifm_data.ifi_opackets, if2m.ifm_data.ifi_ipackets,
118                                     if2m.ifm_data.ifi_obytes, if2m.ifm_data.ifi_ibytes, if2m.ifm_data.ifi_oerrors,
119                                     if2m.ifm_data.ifi_ierrors, if2m.ifm_data.ifi_collisions, if2m.ifm_data.ifi_iqdrops,
120                                     if2m.ifm_data.ifi_baudrate, System.currentTimeMillis()));
121                 }
122             }
123         }
124         return data;
125     }

127     /**
128      * Updates interface network statistics on the given interface. Statistics
129      * include packets and bytes sent and received, and interface speed.
130      *
131      * @param netIF
132      *            The interface on which to update statistics
133      * @return {@code true} if the update was successful, {@code false} otherwise.
134      */
135     public static boolean updateNetworkStats(NetworkIF netIF) {
136         int index = netIF.queryNetworkInterface().getIndex();
137         return updateNetworkStats(netIF, queryIFdata(index));
138     }
139 
140     /**
141      * Updates interface network statistics on the given interface. Statistics
142      * include packets and bytes sent and received, and interface speed.
143      *
144      * @param netIF
145      *            The interface on which to update statistics
146      * @param data
147      *            A map of network interface statistics with the index as the key
148      * @return {@code true} if the update was successful, {@code false} otherwise.
149      */
150     private static boolean updateNetworkStats(NetworkIF netIF, Map<Integer, IFdata> data) {
151         int index = netIF.queryNetworkInterface().getIndex();
152         if (data.containsKey(index)) {
153             IFdata ifData = data.get(index);
154             // Update data
155             netIF.setIfType(ifData.ifType);
156             netIF.setBytesSent(ifData.oBytes);
157             netIF.setBytesRecv(ifData.iBytes);
158             netIF.setPacketsSent(ifData.oPackets);
159             netIF.setPacketsRecv(ifData.iPackets);
160             netIF.setOutErrors(ifData.oErrors);
161             netIF.setInErrors(ifData.iErrors);
162             netIF.setCollisions(ifData.collisions);
163             netIF.setInDrops(ifData.iDrops);
164             netIF.setSpeed(ifData.speed);
165             netIF.setTimeStamp(ifData.timeStamp);
166             return true;
167         }
168         return false;
169     }
