38  public interface Wtsapi32 extends com.sun.jna.platform.win32.Wtsapi32 {
39  
40      Wtsapi32 INSTANCE = Native.load("Wtsapi32", Wtsapi32.class, W32APIOptions.DEFAULT_OPTIONS);

42      int DOMAIN_LENGTH = 17;
43      int USERNAME_LENGTH = 20;
44      int WINSTATIONNAME_LENGTH = 32;
45  
46      /**
47       * Contains information about a client session on a Remote Desktop Session Host
48       * (RD Session Host) server.
49       */
50      @FieldOrder({ "SessionId", "pWinStationName", "State" })
51      class WTS_SESSION_INFO extends Structure {
52          public int SessionId;
53          public String pWinStationName; // either LPSTR or LPWSTR
54          public int State; // WTS_CONNECTSTATE_CLASS
55  
56          public WTS_SESSION_INFO() {
57              super(W32APITypeMapper.DEFAULT);
58          }
59  
60          public WTS_SESSION_INFO(Pointer p) {
61              super(p, Structure.ALIGN_DEFAULT, W32APITypeMapper.DEFAULT);
62              read();
63          }
64      }
65  
66      /**
67       * Contains the client network address of a Remote Desktop Services session.
68       */
69      @FieldOrder({ "AddressFamily", "Address" })
70      class WTS_CLIENT_ADDRESS extends Structure {
71          public int AddressFamily;
72          public byte[] Address = new byte[20];
73  
74          public WTS_CLIENT_ADDRESS() {
75              super();
76          }
77  
78          public WTS_CLIENT_ADDRESS(Pointer p) {
79              super(p);
80              read();
81          }
82      }

84      /**
85       * Contains information about a Remote Desktop Services session.
86       */
87      @FieldOrder({ "State", "SessionId", "IncomingBytes", "OutgoingBytes", "IncomingFrames", "OutgoingFrames",
88              "IncomingCompressedBytes", "OutgoingCompressedBytes", "WinStationName", "Domain", "UserName", "ConnectTime",
89              "DisconnectTime", "LastInputTime", "LogonTime", "CurrentTime" })
90      class WTSINFO extends Structure {
91          private static final int CHAR_WIDTH = Boolean.getBoolean("w32.ascii") ? 1 : 2;
92  
93          public int State;
94          public int SessionId;
95          public int IncomingBytes;
96          public int OutgoingBytes;
97          public int IncomingFrames;
98          public int OutgoingFrames;
99          public int IncomingCompressedBytes;
100         public int OutgoingCompressedBytes;
101         public byte[] WinStationName = new byte[WINSTATIONNAME_LENGTH * CHAR_WIDTH];
102         public byte[] Domain = new byte[DOMAIN_LENGTH * CHAR_WIDTH];
103         public byte[] UserName = new byte[(USERNAME_LENGTH + 1) * CHAR_WIDTH];
104         public LARGE_INTEGER ConnectTime;
105         public LARGE_INTEGER DisconnectTime;
106         public LARGE_INTEGER LastInputTime;
107         public LARGE_INTEGER LogonTime;
108         public LARGE_INTEGER CurrentTime;
109 
110         public WTSINFO() {
111             super(W32APITypeMapper.DEFAULT);
112         }
113 
114         public WTSINFO(Pointer p) {
115             super(p, Structure.ALIGN_DEFAULT, W32APITypeMapper.DEFAULT);
116             read();
117         }
118 
119         public String getWinStationName() {
120             return getStringAtOffset(fieldOffset("WinStationName"));
121         }
122 
123         public String getDomain() {
124             return getStringAtOffset(fieldOffset("Domain"));
125         }
126 
127         public String getUserName() {
128             return getStringAtOffset(fieldOffset("UserName"));
129         }
130 
131         private String getStringAtOffset(int offset) {
132             return CHAR_WIDTH == 1 ? getPointer().getString(offset) : getPointer().getWideString(offset);
133         }
134     }

136     /**
137      * Retrieves a list of sessions on a Remote Desktop Session Host (RD Session
138      * Host) server.
139      *
140      * @param hServer
141      *            A handle to the RD Session Host server.
142      *            <p>
143      *            You can use the {@code WTSOpenServer} or {@code WTSOpenServerEx}
144      *            functions to retrieve a handle to a specific server, or
145      *            {@link #WTS_CURRENT_SERVER_HANDLE} to use the RD Session Host
146      *            server that hosts your application.
147      * @param Reserved
148      *            This parameter is reserved. It must be zero.
149      * @param Version
150      *            The version of the enumeration request. This parameter must be 1.
151      * @param ppSessionInfo
152      *            A pointer to an array of {@link WTS_SESSION_INFO} structures that
153      *            represent the retrieved sessions. To free the returned buffer,
154      *            call the {@link Wtsapi32#WTSFreeMemory} function.
155      * @param pCount
156      *            A pointer to the number of {@code WTS_SESSION_INFO} structures
157      *            returned in the {@code ppSessionInfo} parameter.
158      * @return Returns {@code false} if this function fails. If this function
159      *         succeeds, returns {@code true}.
160      *         <p>
161      *         To get extended error information, call
162      *         {@link Kernel32#GetLastError()}.
163      */
164     boolean WTSEnumerateSessions(HANDLE hServer, int Reserved, int Version, PointerByReference ppSessionInfo,
165             IntByReference pCount);
166 
167     /**
168      * Retrieves session information for the specified session on the specified
169      * Remote Desktop Session Host (RD Session Host) server. It can be used to query
170      * session information on local and remote RD Session Host servers.
171      *
172      * @param hServer
173      *            A handle to an RD Session Host server. Specify a handle opened by
174      *            the {@code WTSOpenServer} function, or specify
175      *            {@link #WTS_CURRENT_SERVER_HANDLE} to indicate the RD Session Host
176      *            server on which your application is running.
177      * @param SessionId
178      *            A Remote Desktop Services session identifier. To indicate the
179      *            session in which the calling application is running (or the
180      *            current session) specify {@link #WTS_CURRENT_SESSION}. Only
181      *            specify {@code WTS_CURRENT_SESSION} when obtaining session
182      *            information on the local server. If {@code WTS_CURRENT_SESSION} is
183      *            specified when querying session information on a remote server,
184      *            the returned session information will be inconsistent. Do not use
185      *            the returned data.
186      *            <p>
187      *            You can use the {@code WTSEnumerateSessionsEx} function to
188      *            retrieve the identifiers of all sessions on a specified RD Session
189      *            Host server.
190      *            <p>
191      *            To query information for another user's session, you must have
192      *            Query Information permission.
193      * @param WTSInfoClass
194      *            A value of the {@code WTS_INFO_CLASS} enumeration that indicates
195      *            the type of session information to retrieve in a call to the
196      *            {@code WTSQuerySessionInformation} function.
197      * @param ppBuffer
198      *            A pointer to a variable that receives a pointer to the requested
199      *            information. The format and contents of the data depend on the
200      *            information class specified in the {@code WTSInfoClass} parameter.
201      *            To free the returned buffer, call the {@link #WTSFreeMemory}
202      *            function.
203      * @param pBytesReturned
204      *            A pointer to a variable that receives the size, in bytes, of the
205      *            data returned in ppBuffer.
206      * @return If the function succeeds, returns {@code true}.
207      *         <p>
208      *         If the function fails, returns {@code false}. To get extended error
209      *         information, call {@link Kernel32#GetLastError()}.
210      */
211     boolean WTSQuerySessionInformation(HANDLE hServer, int SessionId, int WTSInfoClass, PointerByReference ppBuffer,
212             IntByReference pBytesReturned);
213 
214     /**
215      * Frees memory allocated by a Remote Desktop Services function.
216      *
217      * @param pMemory
218      *            Pointer to the memory to free.
219      */
220     void WTSFreeMemory(Pointer pMemory);
