24  package oshi.software.os;
25  
26  import oshi.annotation.concurrent.ThreadSafe;
27  
28  /**
29   * Includes key statistics of TCP and UDP protocols
30   */
31  @ThreadSafe
32  public interface InternetProtocolStats {
33  
34      /**
35       * Get the TCP stats for IPv4 connections.
36       * <p>
37       * On macOS connection information requires elevated permissions. Without
38       * elevatd permissions, segment data is estimated.
39       *
40       * @return a {@link TcpStats} object encapsulating the stats.
41       */
42      TcpStats getTCPv4Stats();
43  
44      /**
45       * Get the TCP stats for IPv6 connections, if available. If not available
46       * separately, these may be 0 and included in IPv4 connections.
47       *
48       * @return a {@link TcpStats} object encapsulating the stats.
49       */
50      TcpStats getTCPv6Stats();
51  
52      /**
53       * Get the UDP stats for IPv4 datagrams.
54       *
55       * @return a {@link UdpStats} object encapsulating the stats.
56       */
57      UdpStats getUDPv4Stats();
58  
59      /**
60       * Get the UDP stats for IPv6 datagrams, if available. If not available
61       * separately, these may be 0 and included in IPv4 datagrams.
62       *
63       * @return a {@link UdpStats} object encapsulating the stats.
64       */
65      UdpStats getUDPv6Stats();
66  
67      final class TcpStats {
68          private final long connectionsEstablished;
69          private final long connectionsActive;
70          private final long connectionsPassive;
71          private final long connectionFailures;
72          private final long connectionsReset;
73          private final long segmentsSent;
74          private final long segmentsReceived;
75          private final long segmentsRetransmitted;
76          private final long inErrors;
77          private final long outResets;
78  
79          public TcpStats(long connectionsEstablished, long connectionsActive, long connectionsPassive,
80                  long connectionFailures, long connectionsReset, long segmentsSent, long segmentsReceived,
81                  long segmentsRetransmitted, long inErrors, long outResets) {
82              this.connectionsEstablished = connectionsEstablished;
83              this.connectionsActive = connectionsActive;
84              this.connectionsPassive = connectionsPassive;
85              this.connectionFailures = connectionFailures;
86              this.connectionsReset = connectionsReset;
87              this.segmentsSent = segmentsSent;
88              this.segmentsReceived = segmentsReceived;
89              this.segmentsRetransmitted = segmentsRetransmitted;
90              this.inErrors = inErrors;
91              this.outResets = outResets;
92          }
93  
94          /**
95           * Connections Established is the number of TCP connections for which the
96           * current state is either ESTABLISHED or CLOSE-WAIT
97           *
98           * @return the connectionsEstablished
99           */
100         public long getConnectionsEstablished() {
101             return connectionsEstablished;
102         }
103 
104         /**
105          * Connections Active is the number of times TCP connections have made a direct
106          * transition to the SYN-SENT state from the CLOSED state. In other words, it
107          * shows a number of connections which are initiated by the local computer. The
108          * value is a cumulative total.
109          *
110          * @return the connectionsActive
111          */
112         public long getConnectionsActive() {
113             return connectionsActive;
114         }
115 
116         /**
117          * Connections Passive is the number of times TCP connections have made a direct
118          * transition to the SYN-RCVD state from the LISTEN state. In other words, it
119          * shows a number of connections to the local computer, which are initiated by
120          * remote computers. The value is a cumulative total.
121          *
122          * @return the connectionsPassive
123          */
124         public long getConnectionsPassive() {
125             return connectionsPassive;
126         }
127 
128         /**
129          * Connection Failures is the number of times TCP connections have made a direct
130          * transition to the CLOSED state from the SYN-SENT state or the SYN-RCVD state,
131          * plus the number of times TCP connections have made a direct transition to the
132          * LISTEN state from the SYN-RCVD state.
133          *
134          * @return the connectionFailures
135          */
136         public long getConnectionFailures() {
137             return connectionFailures;
138         }
139 
140         /**
141          * Connections Reset is the number of times TCP connections have made a direct
142          * transition to the CLOSED state from either the ESTABLISHED state or the
143          * CLOSE-WAIT state.
144          *
145          * @return the connectionsReset
146          */
147         public long getConnectionsReset() {
148             return connectionsReset;
149         }
150 
151         /**
152          * Segments Sent is the number of segments sent, including those on current
153          * connections, but excluding those containing only retransmitted bytes.
154          *
155          * @return the segmentsSent
156          */
157         public long getSegmentsSent() {
158             return segmentsSent;
159         }
160 
161         /**
162          * Segments Received is the number of segments received, including those
163          * received in error. This count includes segments received on currently
164          * established connections.
165          *
166          * @return the segmentsReceived
167          */
168         public long getSegmentsReceived() {
169             return segmentsReceived;
170         }
171 
172         /**
173          * Segments Retransmitted is the number of segments retransmitted, that is,
174          * segments transmitted containing one or more previously transmitted bytes.
175          *
176          * @return the segmentsRetransmitted
177          */
178         public long getSegmentsRetransmitted() {
179             return segmentsRetransmitted;
180         }
181 
182         /**
183          * The number of errors received.
184          *
185          * @return the inErrors
186          */
187         public long getInErrors() {
188             return inErrors;
189         }
190 
191         /**
192          * The number of segments transmitted with the reset flag set.
193          *
194          * @return the outResets
195          */
196         public long getOutResets() {
197             return outResets;
198         }
199 
200         @Override
201         public String toString() {
202             return "TcpStats [connectionsEstablished=" + connectionsEstablished + ", connectionsActive="
203                     + connectionsActive + ", connectionsPassive=" + connectionsPassive + ", connectionFailures="
204                     + connectionFailures + ", connectionsReset=" + connectionsReset + ", segmentsSent=" + segmentsSent
205                     + ", segmentsReceived=" + segmentsReceived + ", segmentsRetransmitted=" + segmentsRetransmitted
206                     + ", inErrors=" + inErrors + ", outResets=" + outResets + "]";
207         }
208     }
209 
210     final class UdpStats {
211         private final long datagramsSent;
212         private final long datagramsReceived;
213         private final long datagramsNoPort;
214         private final long datagramsReceivedErrors;
215 
216         public UdpStats(long datagramsSent, long datagramsReceived, long datagramsNoPort,
217                 long datagramsReceivedErrors) {
218             this.datagramsSent = datagramsSent;
219             this.datagramsReceived = datagramsReceived;
220             this.datagramsNoPort = datagramsNoPort;
221             this.datagramsReceivedErrors = datagramsReceivedErrors;
222         }
223 
224         /**
225          * Datagrams Sent is the number of UDP datagrams sent from the entity.
226          *
227          * @return the datagramsSent
228          */
229         public long getDatagramsSent() {
230             return datagramsSent;
231         }
232 
233         /**
234          * Datagrams Received is the number of UDP datagrams delivered to UDP users.
235          *
236          * @return the datagramsReceived
237          */
238         public long getDatagramsReceived() {
239             return datagramsReceived;
240         }
241 
242         /**
243          * Datagrams No Port is the number of received UDP datagrams for which there was
244          * no application at the destination port.
245          *
246          * @return the datagramsNoPort
247          */
248         public long getDatagramsNoPort() {
249             return datagramsNoPort;
250         }
251 
252         /**
253          * Datagrams Received Errors is the number of received UDP datagrams that could
254          * not be delivered for reasons other than the lack of an application at the
255          * destination port.
256          *
257          * @return the datagramsReceivedErrors
258          */
259         public long getDatagramsReceivedErrors() {
260             return datagramsReceivedErrors;
261         }
262 
263         @Override
264         public String toString() {
265             return "UdpStats [datagramsSent=" + datagramsSent + ", datagramsReceived=" + datagramsReceived
266                     + ", datagramsNoPort=" + datagramsNoPort + ", datagramsReceivedErrors=" + datagramsReceivedErrors
267                     + "]";
268         }
269     }
