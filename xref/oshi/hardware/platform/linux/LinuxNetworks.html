24  package oshi.hardware.platform.linux;
25  
26  import java.io.File;
27  
28  import oshi.annotation.concurrent.ThreadSafe;
29  import oshi.hardware.NetworkIF;
30  import oshi.hardware.common.Networks;
31  import oshi.util.FileUtil;
32  
33  /**
34   * LinuxNetworks class.
35   */
36  @ThreadSafe
37  public final class LinuxNetworks extends Networks {
38      /**
39       * Updates interface network statistics on the given interface. Statistics
40       * include packets and bytes sent and received, and interface speed.
41       *
42       * @param netIF
43       *            The interface on which to update statistics
44       * @return {@code true} if the update was successful, {@code false} otherwise.
45       */
46      public static boolean updateNetworkStats(NetworkIF netIF) {
47          try {
48              File ifDir = new File(String.format("/sys/class/net/%s/statistics", netIF.getName()));
49              if (!ifDir.isDirectory()) {
50                  return false;
51              }
52          } catch (SecurityException e) {
53              return false;
54          }
55          String ifTypePath = String.format("/sys/class/net/%s/type", netIF.getName());
56          String carrierPath = String.format("/sys/class/net/%s/carrier", netIF.getName());
57          String txBytesPath = String.format("/sys/class/net/%s/statistics/tx_bytes", netIF.getName());
58          String rxBytesPath = String.format("/sys/class/net/%s/statistics/rx_bytes", netIF.getName());
59          String txPacketsPath = String.format("/sys/class/net/%s/statistics/tx_packets", netIF.getName());
60          String rxPacketsPath = String.format("/sys/class/net/%s/statistics/rx_packets", netIF.getName());
61          String txErrorsPath = String.format("/sys/class/net/%s/statistics/tx_errors", netIF.getName());
62          String rxErrorsPath = String.format("/sys/class/net/%s/statistics/rx_errors", netIF.getName());
63          String collisionsPath = String.format("/sys/class/net/%s/statistics/collisions", netIF.getName());
64          String rxDropsPath = String.format("/sys/class/net/%s/statistics/rx_dropped", netIF.getName());
65          String speed = String.format("/sys/class/net/%s/speed", netIF.getName());
66  
67          netIF.setTimeStamp(System.currentTimeMillis());
68          netIF.setIfType(FileUtil.getIntFromFile(ifTypePath));
69          netIF.setConnectorPresent(FileUtil.getIntFromFile(carrierPath) > 0);
70          netIF.setBytesSent(FileUtil.getUnsignedLongFromFile(txBytesPath));
71          netIF.setBytesRecv(FileUtil.getUnsignedLongFromFile(rxBytesPath));
72          netIF.setPacketsSent(FileUtil.getUnsignedLongFromFile(txPacketsPath));
73          netIF.setPacketsRecv(FileUtil.getUnsignedLongFromFile(rxPacketsPath));
74          netIF.setOutErrors(FileUtil.getUnsignedLongFromFile(txErrorsPath));
75          netIF.setInErrors(FileUtil.getUnsignedLongFromFile(rxErrorsPath));
76          netIF.setCollisions(FileUtil.getUnsignedLongFromFile(collisionsPath));
77          netIF.setInDrops(FileUtil.getUnsignedLongFromFile(rxDropsPath));
78          long netSpeed = FileUtil.getUnsignedLongFromFile(speed) * 1024 * 1024;
79          netIF.setSpeed(netSpeed < 0 ? 0 : netSpeed);
80          return true;
81      }
82  }
